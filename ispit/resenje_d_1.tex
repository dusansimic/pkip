	\noindent
	Neka je $ n \in \mathbb{N}, n \ge 2 $, i neka su $ a_1, a_2, \ldots, a_n, \in \mathbb{R} $ takvi da
	je $ a_i \ge 0 $ za sve $ i \in \{1, 2, \ldots, n\} $. Tada je:

	\begin{equation*}
		\frac{a_1 + a_2 + \ldots + a_n}{n} \ge \sqrt[n]{a_1 \cdot a_2 \cdot \ldots \cdot a_n}
	\end{equation*}
