	\noindent
	Neka su $ \vec{u} = (x_1, y_1) $ i $ \vec{v} = (x_2, y_2) $ dva vektora u $ \R^2 $ odabrana tako da je $ \vec{u} \neq \vec{0} $ i $ \vec{v} \neq \vec{0} $. Tada je
	\begin{equation*}
		\cos \angle (\vec{u}, \vec{v}) = \frac{\vec{u} \cdot \vec{v}}{|\vec{u}| \cdot |\vec{v}|}.
	\end{equation*}
